## 源代码
* 腐蚀 
**窗口大小的变化依赖调整ii，jj的起始值** 
```C++
cv::Mat* erode(const cv::Mat image) {
	Mat* ret = new Mat(image.size(), CV_8UC1);
	int row = ret->rows;
	int col = ret->cols;
	for (int i = 0; i < row; i++) {
		for (int j = 0; j < col; j++) {
			bool found = false;
			for (int ii = -1; ii <= 1; ii++) {
				if (found)
					break;
				for (int jj = -1; jj <= 1; jj++) {
					int temp_row = i + ii;
					int temp_col = j + jj;
					if (check_position_valid(temp_row, temp_col, row, col)) {
						int cmpval = image.at<uchar>(temp_row, temp_col);
						if (cmpval == 0) {
							found = true;
							break;
						}
					}
				}
				if(found)
					ret->at<uchar>(i, j) = 0;
				else
				{
					ret->at<uchar>(i, j) = 255;
				}
			}
		}
	}
	return ret;
}
```

* 二值化
```C++
cv::Mat* thresholding(const cv::Mat image) {
	Mat* ret = new Mat(image.size(), CV_8UC1);
	int row = ret->rows;
	int col = ret->cols;
	for (int i = 0; i < row; i++) {
		for (int j = 0; j < col; j++) {
			int temp = image.at<uchar>(i, j);
			if (temp > 255 / 2.5) {
				ret->at<uchar>(i, j) = 255;
			}
			else {
				ret->at<uchar>(i, j) = 0;
			}
		}
	}
	return ret;
}
```
